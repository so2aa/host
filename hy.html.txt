  <script>
    const questions = {
      grammar: [],
      vocab: [],
      phrasal: []
    };

    let current = 0;
    let selectedCategory = "";

    function showCategories() {
      document.querySelector(".intro-banner").style.display = "none";
      document.getElementById("categoryPage").style.display = "block";
    }

    function startGame(category) {
      selectedCategory = category;
      current = 0;
      document.getElementById("categoryPage").style.display = "none";
      const gameContainer = document.getElementById("gameContainer");
      gameContainer.className = `container ${category}-bg`;
      gameContainer.style.display = "block";
      document.getElementById("gameTitle").innerText = `Mission ${capitalize(category)} ‚Äì Designed by Ayoub El-Kiri`;
      loadQuestion();
    }

    function loadQuestion() {
      const q = questions[selectedCategory][current];
      document.getElementById("question").innerText = q.question;
      document.getElementById("btnA").innerText = q.options.a;
      document.getElementById("btnB").innerText = q.options.b;
      document.getElementById("btnC").innerText = q.options.c;
      document.getElementById("result").innerText = "";
    }

    function checkAnswer(choice) {
      const correct = questions[selectedCategory][current].correct;
      const result = document.getElementById("result");
      if (choice === correct) {
        result.innerText = "‚úÖ Correct! Excellent work!";
        current++;
        if (current < questions[selectedCategory].length) {
          setTimeout(loadQuestion, 1500);
        } else {
          setTimeout(() => {
            document.body.innerHTML = `
              <div class='quote-page'>
                <h1>üéâ Mission Complete!</h1>
                <p>"Education is the passport to the future, for tomorrow belongs to those who prepare for it today." ‚Äì Malcolm X</p>
              </div>
            `;
          }, 1500);
        }
      } else {
        result.innerText = "‚ùå Incorrect! Try again.";
      }
    }

    function capitalize(word) {
      return word.charAt(0).toUpperCase() + word.slice(1);
    }
  </script>
